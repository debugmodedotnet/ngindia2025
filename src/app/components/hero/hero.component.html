export class NgIndiaConferenceStore {

  apiUrl = 'https://api.ng-india.com';

  conferenceResource = httpResource<IConference[]>(() => ({
    url: this.apiUrl + '/conference-2026',
    method: 'GET',
  }));

  stats = computed(() => {
    if (this.conferenceResource.hasValue()) {
      return this.conferenceResource.value();
    }
    return [];
  });

  loading = computed(() => 
    this.conferenceResource.isLoading()
  );
  
  totalAttendees = computed(() => 400);
  totalSpeakers = computed(() => 17);
  status = computed(() => 
    this.conferenceResource.status()
  );

  refresh() {
    this.conferenceResource.reload();
  }
}
